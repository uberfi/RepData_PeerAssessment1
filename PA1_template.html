mean.imp <- function (dat){
        missing <- is.na(dat)
        n.missing <- sum(missing)
        dat.obs <- dat[!missing]
        imputed <- dat
        imputed[missing] <- mean(dat.obs, n.missing, replace=TRUE)
        return(imputed)
}
X.imp <- mean.imp(X$steps)
X$steps2 <- X.imp
